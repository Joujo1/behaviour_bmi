[Unit]
Description=Cage Controller
After=network-online.target pigpiod.service
Wants=network-online.target pigpiod.service

[Service]
ExecStart=/usr/bin/python3 /home/pi/behaviour_bmi/bmi_closed_loop/RPi_main/main.py
WorkingDirectory=/home/pi/behaviour_bmi/bmi_closed_loop/RPi_main
Restart=always
RestartSec=5
User=root
Environment=PYTHONUNBUFFERED=1
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target


# ── Setup ────────────────────────────────────────────────────────────────────

# Step 1: Enable pigpiod to start at boot (pre-installed on Raspberry Pi OS)
sudo systemctl enable pigpiod

# Step 2: Create the service file
sudo nano /etc/systemd/system/cage_controller.service

# Step 3: Paste the contents above, then save (Ctrl+X → Y → Enter)

# Step 4: Enable and start
sudo systemctl daemon-reload
sudo systemctl enable cage_controller.service
sudo systemctl start cage_controller.service


# ── Useful commands ───────────────────────────────────────────────────────────

# Check status
sudo systemctl status cage_controller.service

# Live logs
journalctl -u cage_controller.service -f

# Restart after code changes
sudo systemctl restart cage_controller.service

# Stop
sudo systemctl stop cage_controller.service
